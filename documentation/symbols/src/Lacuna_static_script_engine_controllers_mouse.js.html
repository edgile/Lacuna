1 var mouseController = function(config){
  2 	if(config && config.engine){
  3 		this.engine = config.engine;
  4 	}
  5 	this.enabled = false;
  6 	this.controllerType = 'absolute'; // absolute (e.g. mouse) or relative (e.g. joystick)
  7 	this.touchable = 'createTouch' in document; // is this running in a touch capable environment?
  8 	if(!this.touchable) { // Disable the mouse controller for devices that support touch
  9 		this.enabled = true;
 10 		if(document.addEventListener) { // Opera - Firefox - Google Chrome
 11 			document.addEventListener("mousedown", this.down.bind(this), false);
 12 			document.addEventListener("mousemove", this.move.bind(this), false);
 13 			document.addEventListener("mouseup", this.up.bind(this), false);
 14 		}
 15 		else if(document.attachEvent) { // Internet Explorer
 16 			document.attachEvent("onmousedown", this.down.bind(this));
 17 			document.attachEvent("onmousemove", this.move.bind(this));
 18 			document.attachEvent("onmouseup", this.up.bind(this));
 19 		}
 20 		else if(!document.onmousedown && !document.onmouseup) {
 21 			document.onmousedown = this.down.bind(this);
 22 			document.onmousemove = this.move.bind(this);
 23 			document.onmouseup = this.up.bind(this);
 24 		} 
 25 	}
 26 }; 
 27 
 28 mouseController.prototype.down = function (e) {  
 29 	this.engine.buttonDown = true;
 30 };
 31 
 32 mouseController.prototype.move = function (e) {  
 33 	this.mousePosition = {x: e.clientX, y: e.clientY};
 34 };
 35 
 36 mouseController.prototype.up = function (e) {  
 37 	this.engine.buttonDown = false;
 38 };
