1 var touchController = function(config){
  2 	if(config && config.engine){
  3 		this.engine = config.engine;
  4 	}
  5 	this.resize();
  6 	this.leftTouchID = -1;
  7 	this.rightTouchID = -1;
  8 	this.secondRightTouchID = -1;
  9 	this.leftTouchPos = {x: 0, y: 0};
 10 	this.leftTouchStartPos = {x: 0, y: 0};
 11 	this.leftVector = {x: 0, y: 0};
 12 	this.touchable = 'createTouch' in document; // is this running in a touch capable environment?
 13 	this.touches = []; // array of touch vectors
 14 	if(this.touchable) {
 15 		window.addEventListener( 'touchstart', this.onTouchStart.bind(this), false );
 16 		window.addEventListener( 'touchmove', this.onTouchMove.bind(this), false );
 17 		window.addEventListener( 'touchend', this.onTouchEnd.bind(this), false );  
 18 	}
 19 }; 
 20 
 21 touchController.prototype.resize = function (e) {  
 22 	this.halfWidth = window.innerWidth / 2; 
 23 	this.halfHeight = window.innerHeight / 2;
 24 };
 25 
 26 touchController.prototype.render = function(c) { // c is the canvas' context 2D
 27 	if(this.touchable) {
 28 		var circleColor = 'rgba(255,255,255,0.3)';
 29 		for(var i = 0, l = this.touches.length; i < l; i++){
 30 			var touch = this.touches[i];
 31 			var circleColor = 'rgba(255,255,255,0.3)';
 32 			if(touch.identifier === this.leftTouchID){
 33 				this.drawCircle(c, circleColor, 6, this.leftTouchStartPos, 40);
 34 				this.drawCircle(c, circleColor, 2, this.leftTouchStartPos, 60);
 35 				this.drawCircle(c, circleColor, 2, this.leftTouchPos, 40); 
 36 			} 
 37 			else {
 38 				this.drawCircle(c, circleColor, 6, {x: touch.clientX, y: touch.clientY}, 50);
 39 			}
 40 		}
 41 	}
 42 };
 43 
 44 touchController.prototype.drawCircle = function(c, color, lineWidth, center, radius) { // c is the canvas' context 2D
 45 	c.beginPath(); 
 46 	c.strokeStyle = color; 
 47 	c.lineWidth = lineWidth; 
 48 	c.arc(center.x, center.y, radius, 0, Math.PI * 2, true); 
 49 	c.stroke();
 50 };

 52 touchController.prototype.onTouchStart = function(e) {
 53 	for(var i = 0, l = e.changedTouches.length; i < l; i++){
 54 		var touch = e.changedTouches[i]; 
 55 		if((this.leftTouchID < 0) && (touch.clientX < this.halfWidth)) {
 56 			this.leftTouchID = touch.identifier; 
 57 			this.leftTouchStartPos = {x: touch.clientX, y: touch.clientY}; 	
 58 			this.leftTouchPos = {x: touch.clientX, y: touch.clientY}; 
 59 			this.leftVector = {x: 0, y: 0};	
 60 		} 
 61 		else if(this.rightTouchID < 0){	
 62 			// Button down event
 63 			this.rightTouchID = touch.identifier;
 64 			this.engine.buttonDown = true;
 65 		}
 66 		else if(this.secondRightTouchID < 0){	
 67 			// Button down event
 68 			this.secondRightTouchID = touch.identifier;
 69 			this.engine.rightButtonDown = true;
 70 		}	
 71 	}
 72 	this.touches = e.touches; 
 73 };
 74 	
 75 touchController.prototype.onTouchMove = function(e) {
 76 	e.preventDefault(); // Prevent the browser from doing its default thing (scroll, zoom)
 77 	for(var i = 0, l = e.changedTouches.length; i < l; i++){
 78 		var touch = e.changedTouches[i]; 
 79 		if(this.leftTouchID === touch.identifier){
 80 			this.leftTouchPos = {x: touch.clientX, y: touch.clientY}; 
 81 			this.leftVector = {x: touch.clientX - this.leftTouchStartPos.x, y: touch.clientY - this.leftTouchStartPos.y};
 82 			this.engine.mousePosition = {
 83 				x: Math.floor(this.leftVector.x / 5),
 84 				y: Math.floor(this.leftVector.y / 5)
 85 			};
 86 			this.engine.absoluteController = false;
 87 			break; 		
 88 		}		
 89 	}
 90 	this.touches = e.touches; 
 91 };
 92 
 93 touchController.prototype.onTouchEnd = function(e) { 
 94 	this.touches = e.touches;
 95 	for(var i = 0, l = e.changedTouches.length; i < l; i++){
 96 		var touch = e.changedTouches[i];
 97 		if(this.leftTouchID === touch.identifier){
 98 			this.leftTouchID = -1; 
 99 			this.leftVector = {x: 0, y: 0};
100 			this.engine.mousePosition = this.leftVector;
101 			this.engine.absoluteController = false; 		
102 		}
103 		else if(this.rightTouchID === touch.identifier){
104 			this.rightTouchID = -1;
105 			this.engine.buttonDown = false;
106 		}
107 		else if(this.secondRightTouchID === touch.identifier){
108 			this.secondRightTouchID = -1;
109 			this.engine.rightButtonDown = false;
110 		}
111 	}
112 };
