1 /**
  2 *   Represents Line segment.
  3 *   @class
  4 *   @param {THREE.Vector2} start - Start point of the line segment.
  5 *   @param {THREE.Vector2} end - End point of the line segment.
  6 */
  7 var LineSegment = function (start, end) {
  8     this.setStart(start);
  9     this.setEnd(end);
 10 };
 11 
 12 /**
 13     Sets the starting point of the LineSegment.
 14     @function
 15     @param {THREE.Vector2} start - Start point of the line segment.
 16 */
 17 LineSegment.prototype.setStart = function (start) {
 18     this.start = start;
 19 };
 20 
 21 /**
 22     Sets the end point of the LineSegment.
 23     @function
 24     @param {THREE.Vector2} end - End point of the line segment.
 25 */
 26 LineSegment.prototype.setEnd = function (end) {
 27     this.end = end;
 28 };
 29 
 30 /**
 31     Gets the length of the LineSegment.
 32     @function
 33     @returns {number} Length of the line segment.
 34 */
 35 LineSegment.prototype.getLength = function () {
 36     if (this.start && this.end) {
 37         return this.start.distanceTo(this.end);
 38     }
 39     return 0;
 40 };
 41 
 42 /**
 43     Gets the intersection point of the given LineSegments.
 44     @function
 45     @param {LineSegment} segment1 First LineSegment
 46     @param {LineSegment} segment2 Second LineSegment
 47     @returns {THREE.Vector2} Returns the intersection point if the LineSegments intersect otherwise returns null.
 48 */
 49 LineSegment.intersect = function (segment1, segment2) {
 50 	var A = segment1.start;
 51 	var B = segment1.end;
 52 	var C = segment2.start;
 53 	var D = segment2.end;
 54 	
 55 	var denominator = (B.x-A.x)*(D.y-C.y)-(B.y-A.y)*(D.x-C.x);
 56 
 57     // Parallel lines
 58     if (denominator == 0) return null;
 59 
 60 	var r = ((A.y-C.y)*(D.x-C.x)-(A.x-C.x)*(D.y-C.y))/denominator;
 61 	var s = ((A.y-C.y)*(B.x-A.x)-(A.x-C.x)*(B.y-A.y))/denominator;
 62 
 63     // Intersection on both line segments?
 64     if (0 <= r && r <= 1 && 0 <= s && s <= 1) {
 65         var result = new THREE.Vector2();
 66         result.x = A.x + r * (B.x-A.x);
 67         result.y = A.y + r * (B.y-A.y);
 68 
 69         return result;
 70     }
 71     return null;
 72 };
 73 
 74 /**
 75     Gets the intersection point with the given LineSegment.
 76     @function
 77     @param {LineSegment} segment LineSegment to test against
 78     @returns {THREE.Vector2} Returns the intersection point if the object intersects with the given LineSegment otherwise returns null.
 79 */
 80 LineSegment.prototype.intersectSelf = function (segment) {
 81     return LineSegment.intersect(this, segment);
 82 };
