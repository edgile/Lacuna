1 /**
  2     Represents a PolyLine.
  3     @class
  4 */
  5 var PolyLine = function () {
  6     this.points = [];
  7 };
  8 
  9 /**
 10     Gets the requested point
 11     @function
 12     @param {integer} index - Index of the item to remove from the list of points.
 13 */
 14 PolyLine.prototype.getPoint = function (index) {
 15     return this.points[index];
 16 };
 17 
 18 /**
 19     Adds a point to the poly line
 20     @function
 21     @param {THREE.Vector2} point - Point to add to the PolyLine.
 22 */
 23 PolyLine.prototype.addPoint = function (point) {
 24     this.points.push(point);
 25 };
 26 
 27 /**
 28     Removes a point from the PolyLine
 29     @function
 30     @param {integer} index - Index of the point to remove from the PolyLine.
 31 */
 32 PolyLine.prototype.removePoint = function (index) {
 33     this.points.splice(index, 1);
 34 };
 35 
 36 /**
 37     Gets the total length of the PolyLine
 38     @function
 39     @returns {number} Total length of the PolyLine.
 40 */
 41 PolyLine.prototype.getLength = function () {
 42     var result = 0;
 43     for (var i = 1, numberOfPoints = this.points.length; i < numberOfPoints; i++) {
 44         result += this.points[i - 1].distanceTo(this.points[i]);
 45     }
 46     return result;
 47 };
