  8 var engine = function(config){
  9 	helpers.apply({
 10 		debug : false,
 11 		width : 800,
 12 		height : 600,
 13 		pageColor : '#555',
 14 		canvasColor : '#000',
 15 		playerCount : 0
 16 	}, this);
 17 	helpers.apply(config, this);
 18 	if(this.mode == 'standalone' || this.mode == 'client'){
 19 		// Disable selection of text/elements in the browser
 20 		document.onselectstart = function() {return false;};
 21 		document.body.style.background = this.pageColor;
 22 		// Initialize the settings UI
 23 		engine.settings.initialize();
 24 		this.initializeControllers();
 25 	}
 26 	this.buttonDown = false;
 27 	this.mousePosition = {x: 0, y: 0};
 28 	this.buttonDown2 = false;
 29 	this.mousePosition2 = {x: 0, y: 0};
 30 	this.entities = [];
 31 	this.remoteData = [];
 32 	this.previousControllerMessageTime = 0;
 33 	this.previousGameStateMessageTime = 0;
 34 	this.remoteDataString = [];
 35 	this.finishedEntities = [];
 36     this.startTime = new Date().getTime();
 37 	if(this.rulesType){
 38 		this.rules = new this.rulesType({engine: this});
 39 		this.add(this.rules);
 40 	}
 41 	if(this.mode === 'server'){
 42 		this.serverUpdateLoop();
 43 	}
 44 	else{		
 45 		// http://paulirish.com/2011/requestanimationframe-for-smart-animating/
 46 		if ( !window.requestAnimationFrame ) {
 47 			window.requestAnimationFrame = ( function() {
 48 				return window.webkitRequestAnimationFrame ||
 49 				window.mozRequestAnimationFrame ||
 50 				window.oRequestAnimationFrame ||
 51 				window.msRequestAnimationFrame ||
 52 				function( /* function FrameRequestCallback */ callback, /* DOMElement Element */ element ) {
 53 					window.setTimeout( callback, 1000 / 60 );
 54 				};
 55 			})();
 56 		}
 57 		var self = this;
 58 		if(this.socket){
 59 			this.socket.on('game state', function(msg){
 60 				self.remoteDataString.push(msg);
 61 			});
 62 			this.socket.on('reset', function(msg){
 63 				self.reset();
 64 			});
 65 		}
 66 		this.animate();
 67 		document.body.oncontextmenu = function(){return false;};

 86 engine.prototype.animate = function(){
 87 	requestAnimationFrame(this.animate.bind(this));
 88 	this.update();
 89 	// If the renderer is configured incorrectly fall back to classic
 90 	if(!engine.rendering[engine.renderer]){
 91 		engine.renderer = 'classic';
 92 	}
 93 	// Check to see if the renderer changed, 
 94 	// if so, suspend the previous renderer (if it exists at all)
 95 	if(engine.previousRenderer && engine.previousRenderer !== engine.renderer){
 96 		engine.rendering[engine.previousRenderer].call(this, true);
 97 	}
 98 	engine.rendering[engine.renderer].call(this);
 99 	engine.previousRenderer = engine.renderer;
100 };

102 engine.prototype.add = function(entity){
103 	if(this.entities.length === 0){
104 		this.entityIdCounter = 0;
105 	}
106 	if(!entity.engineId && this.mode === 'server'){
107 		entity.engineId = this.entityIdCounter;
108 		this.entityIdCounter++;
109 	}
110 	this.entities.push(entity);
111 	entity.engine = this;
112 	if(this.physics && entity.physicsEntity){
113 		this.physics.addSpaceObject(entity);
114 	}
115 };

139 engine.prototype.update = function(){
140 	time = new Date().getTime() - this.startTime;
141 	if(this.mode !== "client"){
142 		if(this.physics){
143 			// Using timelapse (current physics does that) is suboptimal
144 			this.lastTime = this.lastTime || new Date().getTime();
145 			var timeLapse = (new Date().getTime()) - this.lastTime;
146 			this.lastTime = new Date().getTime();
147 			this.physics.update((timeLapse / 20));
148 		}
149 	}
150 	else if(this.mode === "client"){
151 		this.sendControllerStateToServer(time);
152 		this.processRemoteData();
153 	}
154 	// Update
155 	this.updateEntities(time);
156 };

158 engine.prototype.sendControllerStateToServer = function(time){
159 	if(this.socket){
160 		// Send controller state to the server
161 		// Send the message at most 20 times per second
162 		if((time - this.previousControllerMessageTime) >= 50) {
163 			this.previousControllerMessageTime = time;
164 			this.controllerMessage =  this.mousePosition.x + ',' + this.mousePosition.y + ',' + (this.buttonDown ? 1 : 0);
165 			// If the controls have not changed, do not send a message
166 			if(this.controllerMessage !== this.previousControllerMessage){
167 				this.previousControllerMessage = this.controllerMessage;
168 				this.socket.emit('controller state',this.controllerMessage);
169 			}
170 		}
171 	}
172 };

174 engine.prototype.updateEntities = function (time){
175 	for(var i = 0, l = this.entities.length; i < l; i++){
176 		var e = this.entities[i];
177 		if(e && e.update){
178 			e.update(time);
179 		}
180 	}
181 	this.removeFinishedEntities();
182 	this.sendGameStateToClient(time);
183 };

185 engine.prototype.removeFinishedEntities = function(){
186 	// Filter out the objects that indicate they are finished
187 	var oldEntityList = this.entities;
188 	this.entities = [];
189 	this.finishedEntities = this.finishedEntities || [];
190 	for(var i = 0, l = oldEntityList.length; i < l; i++){
191 		var e1 = oldEntityList[i];
192 		if(e1 && !e1.finished){
193 			 this.entities.push(e1);
194 		}
195 		else if(e1 && e1.finished){
196 			if(e1.onRemove){
197 				e1.onRemove();
198 			}
199 			if(this.mode === "server"){
200 				this.finishedEntities.push(e1);
201 			}
202 		}
203 	}
204 };

218 engine.prototype.prepareRemoteData = function(time){
219 	var remoteData = '';
220 	if((this.mode === "server") && (((time - this.previousGameStateMessageTime) >= 50) || (this.previousGameStateMessageTime > time))){
221 		this.previousGameStateMessageTime = time;
222 		remoteData = this.appendRemoteData(this.entities, remoteData);
223 		remoteData = this.appendRemoteData(this.finishedEntities, remoteData);
224 		this.finishedEntities = [];
225 	}
226 	if(remoteData){
227 		remoteData = time + ',' + remoteData;
228 	}
229 	return remoteData;
230 };

232 engine.prototype.appendRemoteData = function(entities, remoteData){
233 	for(var i = 0, l = entities.length; i < l; i++){
234 		var e = entities[i];
235 		if(e && e.getRemoteData){
236 			if(e.finished){
237 				remoteData = remoteData ? remoteData + "," : "";
238 				remoteData =  remoteData + e.engineId + ",-1";
239 			}
240 			else{
241 				var newData = e.getRemoteData();
242 				if(newData){
243 					remoteData = remoteData ? remoteData + "," : "";
244 					remoteData =  remoteData + e.engineId + "," + newData;
245 				}
246 			}
247 		}
248 	}
249 	return remoteData;
250 };

252 engine.prototype.getEntityById = function(engineId){
253 	var result = null;
254 	for(var i = 0, m = this.entities.length; i < m; i++){
255 		if(engineId === this.entities[i].engineId){
256 			result = this.entities[i];
257 			break;
258 		}
259 	}
260 	return result;
261 };

263 engine.prototype.processRemoteData = function(){
264 	var s = this.remoteDataString.shift();
265 	while(s){
266 		var dataFromServer = s.split(",");
267 		var offset = 0;
268 		// First item in the array is always the server time for the data
269 		this.lastServerTimedate = dataFromServer[0];
270 		offset++;
271 		var l = dataFromServer.length;
272 		var i = 0;
273 		while(offset < l){
274 			var e = null;
275 			// For each entity, the first array item contains it's id  
276 			var engineId = parseInt(dataFromServer[offset]);
277 			offset++;
278 			e = this.getEntityById(engineId);
279 			// If the object was not found by Id it needs to be created
280 			if(e === null && dataFromServer[offset] !== "-1"){
281 				// The type of the entity is stored in the second array item
282 				var entityType = engine.remoteRenderer[dataFromServer[offset]];
283 				// Create a new instance and add it to the engine
284 				e = new entityType({
285 					engine: this,
286 					engineId: engineId
287 				});
288 				this.add(e);
289 			}
290 			else if(!e && dataFromServer[offset] === "-1"){
291 				// If the type of the entity is -1 that means the object was finished.
292 				// In this case the object was finished but apparently never reached the
293 				// client in the first place (lived extremely shortly on the server). 
294 				// The offset needs to be increased anyway.
295  				offset++;
296 			}
297 			else if(e && dataFromServer[offset] === "-1"){
298 				// If the type of the entity is -1 that means the object was finished.
299 				offset++;
300 				// By tagging the entity as finished the engine will clean it up
301 				e.finished = true;
302 			}
303 			if(e && !e.finished) {
304 				// Update the existing entity with the remote data
305 				offset = e.renderRemoteData(dataFromServer,offset);
306 			}
307 		}
308 		s = this.remoteDataString.shift();
309 	}
310 };

312 //Stores a value in local storage. Will handle objects, will probably fail when called on the server.
313 engine.setItem = function(key, value) {
314     if (typeof value == "object") {
315         value = JSON.stringify(value);
316     }
317     localStorage.setItem(key, value);
318 };
319 
320 // Gets an object from local storage, if no value for key was found defaultValue is returned;
321 engine.getItem = function(key, defaultValue) {
322     var result = null;
323     if(typeof localStorage != 'undefined'){
324 		result = localStorage.getItem(key);
325 	    // assume it is an object that has been stringified
326 	    if (result && result[0] == "{") {
327 	        result = JSON.parse(result);
328 	    }
329     }
330     return result || defaultValue;
331 };
