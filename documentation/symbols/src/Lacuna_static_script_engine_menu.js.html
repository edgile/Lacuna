1 var menu = function (config) {
  2     helpers.apply({
  3         color: '#AAA',
  4         selectedColor: '#FFF',
  5         mousePlaneOffset: 250,
  6         mousePlane: 'z',
  7         topMost: true,
  8         onmouseup: this.onmouseup.bind(this)
  9     }, this);
 10     helpers.apply(config, this);
 11     this.setItems(this.mainMenu);
 12 };
 13 menu.prototype = new entity();
 14 
 15 menu.prototype.gotoRoot = function () {
 16     this.setItems(this.mainMenu);
 17 };
 18 
 19 menu.prototype.onmouseup = function(entity, hit){
 20 	if(this.ignoreNextButtonUp){
 21 		this.ignoreNextButtonUp = false;
 22 	}
 23 	else{
 24 		if(this.selected && this.selected.onMousePlaneUp){
 25 			this.selected.onMousePlaneUp(this, this);
 26 		}
 27 	}
 28 };
 29 
 30 menu.prototype.setItems = function(items){
 31 	if(this.engine.buttonDown){
 32 		this.ignoreNextButtonUp = true;
 33 	}
 34 	this.currentItems = items;
 35 	this.clear();
 36 	this.texts = [];
 37 	var i = 0;
 38 	for(var s in items){
 39 		var text = s;
 40 		if(items.hasOwnProperty(s)){
 41 			if(items[s].getText){
 42 				text = items[s].getText(this);
 43 			}
 44 			var config = helpers.apply(items[s],{
 45 				font: '50px CBM64',
 46 				size: (items[s].size || 50),
 47 				color: this.color,
 48 				engine: this.engine,
 49 				text: text, 
 50 				position: { 
 51 					x: this.engine.width / 2, 
 52 					y: (this.engine.height / 3) + ( i * 60),
 53 					z: this.mousePlaneOffset
 54 				}
 55 			});
 56 			if(config.submenu){
 57 				var self = this;
 58 				config.onMousePlaneUp = function(entity, hit){
 59 					self.setItems(entity.submenu);
 60 				};
 61 			}
 62 			this.texts.push(config);
 63 			i++;
 64 		}
 65 	}
 66 };
 67 
 68 menu.prototype.show = function (items) {
 69     if (this.finished) {
 70         this.finished = false;
 71         if(!this.engine.contains(this)){
 72         	this.engine.add(this);
 73         }
 74         this.gotoRoot();
 75     }
 76     if(items){
 77     	this.setItems(items);
 78     }
 79 };
 80 
 81 menu.prototype.toggle = function () {
 82     if (this.finished) {
 83         this.finished = false;
 84         if(!this.engine.contains(this)){
 85         	this.engine.add(this);
 86         }
 87         this.gotoRoot();
 88     }
 89     else {
 90         this.finished = true;
 91     }
 92 };
 93 
 94 menu.prototype.hide = function () {
 95     this.finished = true;
 96 };
 97 
 98 menu.prototype.clear = function(){
 99 	if(this.subEntities){
100 		for(var i = 0, l = this.subEntities.length; i < l; i++){
101 			this.subEntities[i].finished = true;
102 		}
103 	}
104 };
105 
106 menu.prototype.select = function(entity){
107 	if(this.selected != entity){
108 		if(this.selected){
109 			this.selected.color = this.color;
110 		}
111 		this.selected = entity;
112 		if(this.selected){
113 			this.selected.color = this.selectedColor;
114 		}
115 	}
116 };
117 
118 menu.prototype.update = function(time){
119 	if(this.mousePosition && this.texts){
120 		// Calculate item on the same height as the mouse
121 		var index = Math.floor( (this.mousePosition.y - (this.engine.height / 3)) / 60 ) + 1;
122 		if(index > -1 && index < this.texts.length && this.texts[index].onMousePlaneUp){
123 			this.select(this.texts[index]);
124 		    if(this.engine.touchController.touchable){
125 		    	this.onmouseup(this.texts[index], true);
126 		    	this.ignoreNextButtonUp = true;
127 		    }
128 		}
129 		else{
130 			this.select(null);
131 		}
132 	}
133 };
