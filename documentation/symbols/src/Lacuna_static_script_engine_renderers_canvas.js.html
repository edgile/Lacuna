  2 Engine.rendering.canvas = function(suspend){
  3 	if(!this.canvasRenderer){
  4 		this.canvasRenderer = new Engine.rendering.canvas.renderer({
  5 			engine: this
  6 		});
  7 	}
  8 	this.canvasRenderer.suspend(suspend);
  9 	if(!this.canvasRenderer.suspended){
 10 		this.canvasRenderer.context.fillStyle = this.backgroundColor;
 11 		this.canvasRenderer.context.fillRect(0, 0, window.innerWidth, window.innerHeight);
 12 		// Render
 13 		var topMost = [];
 14 		for(var i = 0, l = this.entities.length; i < l; i++){
 15 			var e = this.entities[i];
 16 			if(e.hidden){
 17 				// Ignore hidden elements
 18 			}
 19 			else if(e.topMost){
 20 				topMost.push(e);
 21 			}
 22 			else {
 23 				this.canvasRenderer.renderEntity(e);
 24 			}
 25 		}
 26 		for(var i = 0, l = topMost.length; i < l; i++){
 27 			this.canvasRenderer.renderEntity(topMost[i]);
 28 		}
 29 		for(var i = 0, l = this.controllers.length; i < l; i++){
 30 			var controller = this.controllers[i];
 31 			if(controller.render){
 32 				controller.render(this.canvasRenderer.context);
 33 			}
 34 		}
 35 	}
 36 };

 38 // Class for canvas renderer
 39 Engine.rendering.canvas.renderer = function(){
 40 
 41 	var drawingFunctions = {
 42 		'line': function(position, config){
 43 			position = config.position || position;
 44 			this.context.strokeStyle = config.color;
 45 			this.context.lineWidth = Math.ceil(config.width * this.scale);
 46 			this.context.beginPath();
 47 			this.context.moveTo(this.fixX(position, config.start), this.fixY(position, config.start));
 48 			this.context.lineTo(this.fixX(position, config.end), this.fixY(position, config.end));
 49 			this.context.stroke();
 50 		},
 51 		'polyline': function(position,config){
 52 		    var points = config.points;
 53 		    position = config.position || position;
 54 		    if (points.points.length > 1) {
 55 		    	this.context.strokeStyle = "white";
 56 		    	this.context.lineWidth = Math.ceil(config.width * this.scale);
 57 		    	this.context.beginPath();
 58 		    	this.context.moveTo( this.fixX(position, points.points[0]), this.fixY(position,  points.points[0]));
 59 			    for (var i = 1, l = points.points.length; i < l; i++) {
 60 			    	this.context.lineTo( this.fixX(position, points.points[i]), this.fixY(position, points.points[i]));
 61 			    }
 62 			    this.context.stroke();
 63 			    this.context.closePath();
 64 		    }
 65 		},
 66 		'arc': function(position, config){
 67 		    if(config.fill){
 68 		    	drawingFunctions.arcFill.apply(this, [position, config]);
 69 		    }
 70 		    else{
 71 		    	drawingFunctions.arcStroke.apply(this, [position, config]);
 72 		    }
 73 		},
 74 		'circle': function(position, config){
 75 			config.start = 0;
 76 			config.end = Math.PI * 2;
 77 			config.close = true;
 78 			drawingFunctions.arc.apply(this, [position, config]);
 79 		},
 80 		'arcStroke': function(position, config){
 81 			this.context.strokeStyle = config.color;
 82 		    this.context.lineWidth = config.width;
 83 		    this.context.beginPath();
 84 		    this.context.arc(this.fixX(position, config.center), this.fixY(position, config.center), config.radius * this.scale, config.start , config.end, config.close);
 85 		    this.context.closePath();
 86 		    this.context.stroke();
 87 		},
 88 		'arcFill': function(position, config){
 89 			this.context.fillStyle = config.color;
 90 			this.context.beginPath();
 91 			this.context.arc(this.fixX(position, config.center), this.fixY(position, config.center), config.radius * this.scale, config.start , config.end, config.close);
 92 			this.context.fill();
 93 			this.context.closePath();
 94 		}
 95 		
 96 	};

 98 	var renderer = function(config){
 99 		helpers.apply(config, this);
100 		this.canvas = document.createElement('canvas');
101 		document.body.appendChild(this.canvas);
102 		this.canvas.style.position = 'absolute';
103 		this.canvas.style.backgroundColor = '#000';
104 		this.canvas.onmousedown = this.onmousedown.bind(this);
105 		this.canvas.onmouseup = this.onmouseup.bind(this);
106 		window.onmousemove = this.onmousemove.bind(this);
107 		window.onresize = this.resize.bind(this);
108 		this.resize();
109 	};
110 	
111 	renderer.prototype.fixX = function(position, point){
112 		return ((position.x + point.x) * this.scale) + this.offsetLeft;
113 	};
114 	
115 	renderer.prototype.fixY = function(position, point){
116 		return ((position.y + point.y) * this.scale) + this.offsetTop;
117 	};
118 	
119 	renderer.prototype.suspend = function(suspend){
120 		if(this.suspended && !suspend){
121 			this.canvas.style.display = 'block';
122 			window.onmousemove = this.onmousemove.bind(this);
123 			window.onresize = this.resize.bind(this);
124 			this.resize();
125 		}
126 		else if(!this.suspended && suspend){
127 			this.canvas.style.display = 'none';
128 		}
129 		this.suspended = suspend;
130 	};
131 	
132 	renderer.prototype.resize = function(){
133 		//if(this.suspended) return;
134 		this.canvas.style.top = '0px';
135 		this.canvas.style.left = '0px';
136 		this.canvas.width = window.innerWidth;
137 		this.canvas.height = window.innerHeight;
138 		// Calculate optimum scale
139 		var scaleh = this.canvas.width / this.engine.width;
140 		var scalev = this.canvas.height / this.engine.height;
141 		this.scale = (scaleh < scalev) ? scaleh : scalev;
142 		this.offsetTop = Math.ceil((this.canvas.height - (this.engine.height * this.scale)) / 2);
143 		if(this.engine.touchController.touchable && this.offsetTop > 40){
144 			this.offsetTop = 40;
145 		}
146 		this.offsetLeft = Math.ceil((this.canvas.width - (this.engine.width * this.scale)) / 2);
147 		this.context = this.canvas.getContext('2d');
148 		for(var i = 0, l = this.engine.controllers.length; i < l; i++){
149 			var controller = this.engine.controllers[i];
150 			if(controller.render){
151 				controller.resize();
152 			}
153 		}
154 	};
155 	
156 	renderer.prototype.notify = function(eventName, entities){
157 		if(entities && eventName){
158 			for(var i = 0, l = entities.length; i < l; i++ ){
159 	    		if(entities[i][eventName]){
160 	    			entities[i][eventName](entities[i], entities[i]);
161 	    		}
162 	    	}
163 		}
164 	};
165 	
166 	renderer.prototype.onmousemove = function(e){
167 		if(this.suspended) return;
168 		var evt = e || window.event;
169 		var newPosition = {
170 			x: Math.ceil((evt.clientX - this.offsetLeft) / this.scale),
171 			y: Math.ceil((evt.clientY - this.offsetTop) / this.scale)
172 		};
173 		if(!this.engine.touchController.touchable){
174 			this.engine.mousePosition = newPosition;
175 			this.engine.absoluteController = true;
176 		}
177 		var items = this.engine.entities;
178 		this.mouseEntities = [];
179 	    for(var i = 0, l = items.length; i < l; i++){
180 	    	var item = items[i];
181 	    	if(item.mousePlane){
182 	    		item.mousePosition = newPosition;
183 	    		this.mouseEntities.push(item);
184 	    	}
185 	    }
186 	    this.notify('onmousemove', this.mouseEntities);
187 	};
188 	
189 	renderer.prototype.onmousedown = function(){
190 		if(this.suspended) return;
191 		if(!this.engine.touchController.touchable){
192 			this.engine.buttonDown = true;
193 		}
194 		this.notify('onmousedown', this.mouseEntities);
195 	};
196 	
197 	renderer.prototype.onmouseup = function(){
198 		if(this.suspended) return;
199 		if(!this.engine.touchController.touchable){
200 			this.engine.buttonDown = false;
201 		}
202 		this.notify('onmouseup', this.mouseEntities);
203 	};
204 	
205 	renderer.prototype.renderEntity = function(e){
206 		if(this.suspended) return;
207 		if(e.render){
208 			e.render();
209 		}
210 		if(e.rects) {
211 			this.drawRects(helpers.ceilPoint({x: e.position.x, y: e.position.y}), e.rects, e.color, true);
212 		}
213 		if(e.shapes){
214 			for(var j = 0, l = e.shapes.length; j < l; j++ ){
215 				var s = e.shapes[j];
216 				drawingFunctions[s.type].apply(this, [e.position, s]);
217 			}
218 		}
219 		if(e.texts){
220 			for(var j = 0, l = e.texts.length; j < l; j++ ){
221 				var t = e.texts[j];
222 				this.context.font = Math.ceil((t.size || 50) * this.scale) + 'px HeavyDataRegular';
223 				this.context.textAlign = 'center';
224 				this.context.fillStyle = t.color;
225 				this.context.fillText(t.text, Math.ceil(t.position.x * this.scale) + this.offsetLeft, Math.ceil(t.position.y * this.scale) + this.offsetTop);
226 			}
227 		}
228 	};
229 	
230 	renderer.prototype.drawRects = function(offset, rects, color, fill){
231 		if(rects){
232 			for(var i = 0, l = rects.length; i < l; i++){
233 				this.drawRect(offset, rects[i], color, fill);
234 			}
235 		}
236 	};
237 	
238 	renderer.prototype.drawRect = function(offset, rect, color, fill){
239 		if(fill){
240 			this.context.fillStyle = color;
241 			this.context.fillRect(
242 				Math.ceil((rect.x + offset.x) * this.scale) + this.offsetLeft ,
243 				Math.ceil((rect.y + offset.y) * this.scale) + this.offsetTop  ,
244 				Math.ceil(rect.w * this.scale) ,
245 				Math.ceil(rect.h * this.scale) 
246 			);
247 		}
248 		else {
249 			this.context.strokeStyle = color;
250 			this.context.strokeRect(
251 				Math.ceil((rect.x + offset.x) * this.scale) ,
252 				Math.ceil((rect.y + offset.y) * this.scale) ,
253 				Math.ceil(rect.w * this.scale) ,
254 				Math.ceil(rect.h * this.scale) 
255 			);
256 		}
257 	};
258 	
259 	return renderer;
260 }();
