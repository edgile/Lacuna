  4 engine.rendering.webgl = function(suspend){
  5 	if(!this.webglRenderer){
  6 		this.webglRenderer = true;
  7 		var self = this;
  8 		this.webglRenderer = new engine.rendering.webgl.webglRenderer({
  9 			models: [
 10 				'resources/models/ship.js', 
 11 				'resources/models/star.js',
 12 				'resources/models/ufo1.js',
 13 				'resources/models/ufo2.js',
 14 				'resources/models/ufo3.js',
 15 				'resources/models/laser.js',
 16 				'resources/models/cube.js'
 17 	        ],
 18 			centerOffset: {x: this.width / 2, y: this.height / 2},
 19 			engine: this,
 20 			callback: function(){
 21 				self.rendererInitialized = true;
 22 			}
 23 		});
 24 	}
 25 	this.webglRenderer.suspend(suspend);
 26 	if(!this.webglRenderer.suspended){
 27 		if(this.rendererInitialized){
 28 			// Render
 29 			for(var i = 0, l = this.entities.length; i < l; i++){
 30 				var e = this.entities[i];			
 31 				this.webglRenderer.renderEntity(e);
 32 			}
 33 			this.webglRenderer.renderScene();
 34 		}
 35 	}
 36 };

100 	webglRenderer.prototype.suspend = function(suspend){
101 		if(this.suspended && !suspend){
102 			this.canvas.style.display = 'block';
103 		    window.onmousemove = this.onmousemove.bind(this);
104 		    window.onresize = this.resize.bind(this);
105 			this.resize();
106 		}
107 		else if(!this.suspended && suspend){
108 			this.canvas.style.display = 'none';
109 		}
110 		this.suspended = suspend;
111 	};

126 		for(var i = 0, l = entities.length; i < l; i++ ){
127     		var mesh = entities[i].object;
128     		if(mesh && mesh.gameEntity && mesh.gameEntity[eventName]){
129     			mesh.gameEntity[eventName](mesh.gameEntity, entities[i]);
130     		}
131     		else if(entities[i][eventName]){
132     			entities[i][eventName](entities[i], entities[i]);
133     		}
134     	}

137 	webglRenderer.prototype.onmousemove = function(e){
138 		if(this.suspended) return;
139 		var mouse_pos = new THREE.Vector2(e.clientX, e.clientY);
140 		var ray = this.create_mouse_ray(mouse_pos);
141 		this.worldMouse = this.planeIntersect(ray, 'z', 0);
142 		if(!this.engine.touchController.touchable){
143 			this.engine.mousePosition = this.toGamePoint(this.worldMouse);
144 			this.engine.absoluteController = true;
145 		}
146 	    this.mouseEntities = ray.intersectObjects(this.scene.children);
147 	    var items = this.engine.entities;
148 	    for(var i = 0, l = items.length; i < l; i++){
149 	    	var item = items[i];
150 	    	if(item.mousePlane){
151 	    		item.mousePosition = this.toGamePoint(this.planeIntersect(ray, item.mousePlane, item.mousePlaneOffset || 0));
152 	    		this.mouseEntities.push(item);
153 	    	}
154 	    }
155 	    this.notify('onmousemove', this.mouseEntities);
156 	};

161 			y: (this.engine.height - Math.ceil((point.y / this.gameScale) + this.centerOffset.y)),
162 			z: point.z
163 		};

192 	webglRenderer.prototype.pick = function(mouse_pos, obj) {
193 	    var ray = this.create_mouse_ray(mouse_pos);
194 	    if (obj !== undefined) {
195 	        return ray.intersectObject(obj);
196 	    }
197 	    else {
198 	        // return this.scene.intersect(ray);
199 	    	return ray.intersectObjects(this.scene.objects);
200 	    }
201 	};

203 	webglRenderer.prototype.create_mouse_ray = function(mouse_pos) {
204 	    var projector = new THREE.Projector();
205 	    var norm_pos = new THREE.Vector2((mouse_pos.x / window.innerWidth) * 2 - 1, -(mouse_pos.y / window.innerHeight) * 2 + 1);
206 	    var mouse_3d = projector.unprojectVector(new THREE.Vector3(norm_pos.x, norm_pos.y, 1.0), this.camera);
207 	    return new THREE.Ray(this.camera.position, mouse_3d.subSelf(this.camera.position).normalize());
208 	};

210 	webglRenderer.prototype.planeIntersect = function(ray, plane /* x, y z*/, planeOffset){
211 		// planeOffset = start[plane] + (vector[plane] * time)
212 		// -(vector[plane] * time) = start[plane] - planeOffset
213 		// vector[plane] * time = -start[plane] + planeOffset
214 		// time = (-start[plane] + planeOffset) / vector[plane]
215 		// time = (planeOffset - start[plane]) / vector[plane]
216 		var result = null;
217 		var start = ray.origin;
218 		var vector = ray.direction;
219 		if(vector[plane]){
220 			var time = (planeOffset - start[plane]) / vector[plane];
221 			var result = {
222 				x: start.x + (vector.x * time),
223 				y: start.y + (vector.y * time),
224 				z: start.z + (vector.z * time)
225 			};
226 		}
227 		return result;
228 		// 250 = 10 + (5 * time)
229 		// -(5 * time) = 10 - 250
230 		// time = 48
231 	};

252 	webglRenderer.prototype.processTexts = function(e){
253 		if(e.texts){
254 			for(var i = 0, l = e.texts.length; i < l; i++ ){
255 				var t = e.texts[i];
256 				var text = t.text;
257 				if(!t.entity){
258 					t.entity = helpers.apply(t ,{
259 						text : text,
260 						modelIndex: text,
261 						score: 0,
262 						geometry: this.to3dText( text ),
263 						modelScale: 1,
264 						color: t.color,
265 						position: { 
266 							x: t.position.x || 0, 
267 							y: t.position.y || 0,
268 							z: t.position.z || 0
269 						},
270 						rotation: {
271 							x: 0, y:0, z:0
272 						},
273 						direction: -1
274 					});
275 					if(!e.subEntities){
276 						e.subEntities = [];
277 					}
278 					e.subEntities.push(t.entity);
279 				}
280 				if(t.entity.text != text){
281 					t.entity.text = text;
282 					t.entity.mesh = null;
283 					t.entity.modelIndex = text;
284 					t.entity.geometry = this.to3dText( text );
285 				}
286 				t.entity.color = t.color;
287 			}
288 		}
289 	};

292 		var result = null;
293 		if(this.engine.mode == 'standalone' || this.engine.mode == 'client'){
294 			result = (text === '') ? new THREE.Geometry() : new THREE.TextGeometry( text, {
295 				size: 90,
296 				height: 30,
297 				curveSegments: 1,
298 				font: "cbm-64"
299 			});
300 		}
301 		return result;
302 	};

304 	webglRenderer.prototype.setEntityMesh = function(e){
305 		var index = e.modelIndex || 0;
306 		var scale = e.modelScale || this.modelScale;
307 		e.meshCache = e.meshCache || {};
308 		if(e.mesh){
309 			this.scene.remove( e.mesh );
310 		}
311 		if(e.meshCache[e.modelIndex]){
312 			e.mesh = e.meshCache[e.modelIndex];
313 		}
314 		else {
315 			var geometry = e.geometry || this.geometries[index];
316 			e.meshCache[e.modelIndex] = e.mesh = new THREE.Mesh( 
317 				geometry,
318 				this.createMaterial(e.color)
319 			);
320 			e.mesh.gameEntity = e;
321 			e.mesh.gameMesh = true;
322 			e.mesh.gameColor = e.color;
323 			e.mesh.modelIndex = e.modelIndex;
324 			e.mesh.scale.set( scale, scale, scale );
325 			e.mesh.doubleSided = true;
326 		}
327 		this.scene.add( e.mesh );
328 	};

330 	webglRenderer.prototype.renderEntity = function(e){
331 		if(this.suspended) return;
332 		if(!e.mesh){
333 			this.setEntityMesh(e);
334 		}
335 		if(e.mesh && e.position){
336 			if(e.mesh.modelIndex !== e.modelIndex){
337 				this.setEntityMesh(e);
338 			}
339 			if(e.mesh.gameColor !== e.color){
340 				e.mesh.material = this.createMaterial(e.color);
341 				e.mesh.gameColor = e.color;
342 			}
343 			e.mesh.position.x = (e.position.x - this.centerOffset.x) * this.gameScale ;
344 			e.mesh.position.y = (e.position.y - this.centerOffset.y) * -this.gameScale ;
345 			e.mesh.position.z = e.position.z || 0;
346 			if(e.rotation){
347 				e.mesh.rotation.x = e.rotation.x;
348 				e.mesh.rotation.y = e.rotation.y;
349 				e.mesh.rotation.z = e.rotation.z;
350 			}
351 			else{
352 				e.mesh.rotation.y = Math.PI * (e.direction === 1 ? 1.5 : 0.5);
353 			}
354 			e.mesh.lastFrameNumber = this.lastFrameNumber;
355 		}
356 		this.processTexts(e);
357 		if(e.subEntities){
358 			var unfinishedEntities = [];
359 			for(var i = 0, l = e.subEntities.length; i < l; i++){
360 				if(!e.subEntities[i].finished){
361 					unfinishedEntities.push(e.subEntities[i]);
362 					this.renderEntity(e.subEntities[i]);
363 				}
364 			}
365 			e.subEntities = unfinishedEntities;
366 		}
367 		if(e.render){
368 			e.render();
369 		}
370 	};

372 	webglRenderer.prototype.renderScene = function(){
373 		if(this.suspended) return;
374 	    if(this.backgroundPlane.gameColor !== this.engine.canvasColor){
375 	    	this.backgroundPlane.materials[0] = this.createMaterial(this.engine.canvasColor);
376 	    	this.backgroundPlane.gameColor = this.engine.canvasColor;
377 		}
378 		var children = this.scene.children.slice(0);
379 		for(var i = 0, l = children.length; i < l; i++){
380 			var m = children[i];
381 			if(m.gameMesh && m.lastFrameNumber !== this.lastFrameNumber){
382 				this.scene.remove(m);
383 			}
384 		}
385 		if(this.worldMouse){
386 			this.camera.position.x = this.worldMouse.x;
387 			this.camera.position.y = this.worldMouse.y;
388 			this.camera.lookAt({x: 0 , y: 0 , z: 0});
389 			this.spotlight.position.x = this.worldMouse.x / 2;
390 			this.spotlight.position.y = this.worldMouse.y / 2;
391 			this.spotlight.lookAt({x: 0 , y: 0 , z: 0});
392 		}
393 		this.renderer.render( this.scene, this.camera );
394 		//this.composer.render();
395 		this.lastFrameNumber++;
396 		
397 	};
