 69 rules.prototype.parseControlString = function(s, buttonName, positionName){

 78 rules.prototype.update = function(time){
 79 	if(!this.initialized){
 80 		this.initialize();
 81 	}
 82 	if(this.engine.mode !== 'server'){
 83 		if(this.engine.gameState.gameOver !== this.previousGameOverState){
 84 			this.previousGameOverState = this.engine.gameState.gameOver;
 85 			if(this.engine.gameState.gameOver && (this.engine.playerCount > 0 || this.engine.mode === 'client')){
 86 				this.showGameOver();
 87 			}
 88 			else if(this.engine.gameState.gameOver && this.engine.playerCount === 0){
 89 				if(this.menu.finished){
 90 					this.engine.reset();
 91 				}
 92 				else{
 93 					this.engine.reset(this.menu);
 94 				}
 95 				return;
 96 			}
 97 			else if(this.engine.mode === 'client'){
 98 				this.hideMenu();
 99 			}
100 		}
101 		this.scoreBar.setScore(this.engine.gameState.player1Score, this.engine.gameState.player2Score);
102 		if(this.previousRightButtonDown !== this.engine.rightButtonDown){
103 			if(this.engine.rightButtonDown === false){
104 				this.toggleMenu();
105 			}
106 			this.previousRightButtonDown = this.engine.rightButtonDown;
107 		}
108 	}

116 rules.prototype.getRemoteData = function(){
117 	var result = null;
118 	var newMessage = "5," + this.engine.gameState.player1Score + "," +
119 	        this.engine.gameState.player2Score + "," +
120 	        (this.engine.gameState.gameOver ? 1 : 0);
121 	if(newMessage !== this.previousMessage){
122 		result = this.previousMessage = newMessage;
123 	}
124 	return result;
125 };

190 rules.prototype.keyboardHandler = function (keyCode) {
191     // 1: Restart game single player mode, standalone (all runs on the client)
192     if (keyCode == 49) {
193         this.startSinglePlayerGame();
194     }
195     // 2: restart game multi player mode, 'client' only renders on the client, game logic runs on the server
196     if (keyCode == 50) {
197     	this.startMultiPlayerGame();
198     }
199     // 3: Audio volume down
200     if (keyCode == 51) {
201         audio.decreaseVolume();
202     }
203     // 4: Audio volume up
204     if (keyCode == 52) {
205         audio.increaseVolume();
206     }
207     // 5: Mute
208     if (keyCode == 53) {
209         audio.mute();
210     }
211     // 6: Restart game in zero player mode
212     if (keyCode == 54) {
213     	this.startZeroPlayerGame();
214     }
215     // M: Toggle menu
216     if (keyCode === 77) {
217 		this.toggleMenu();
218     }
219 };
