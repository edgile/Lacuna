1 /**
  2  * Score object keeps track of the score for the level.
  3  * @param config
  4  * @class
  5  */
  6 function Score(config){
  7 	helpers.apply(config, this);
  8 	
  9 	this.totalPoints = 0;
 10 	this.shipsLanded = 0;
 11 	this.initialized = false;
 12 };
 13 
 14 /**
 15  * Initializes the Score object
 16  * @function
 17  * @private
 18  */
 19 Score.prototype.initialize = function(){
 20 	if(!this.initialized) {
 21 		this.scoreBar = this.level.engine.flow.scoreBar;
 22 		if(!this.scoreBar.hasItem("totalscore")){
 23 			this.scoreBar.addScoreBarItem("totalscore", "Total: 0", "white", 10);
 24 		}
 25 		if(!this.scoreBar.hasItem("shipslanded")){
 26 			this.scoreBar.addScoreBarItem("shipslanded", "Landed: 0", "white", 200);
 27 		}
 28 		this.scoreBar.setScore("totalscore", this.totalPoints);
 29 		this.scoreBar.setScore("shipslanded", "Landed: " + this.shipsLanded);
 30 		this.initialized = true;
 31 	}
 32 };
 33 
 34 Score.prototype.getResult = function(){
 35 	if(this.shipsLanded > 0){
 36 		return {passed: true, text: "Passed" };
 37 	}else{
 38 		return {passed: false, text: "Failed" };
 39 	}
 40 };
 41 
 42 Score.prototype.shipLanded = function(){
 43 	this.shipsLanded += 1;
 44 	if(this.scoreBar){
 45 		this.scoreBar.setScore("shipslanded", "Landed:" + Math.floor(this.shipsLanded));
 46 	}
 47 };
 48 
 49 Score.prototype.addPoints = function(amount){
 50 	this.initialize();
 51 	
 52 	this.totalPoints += amount;
 53 	if(this.scoreBar){
 54 		this.scoreBar.setScore("totalscore", "" + Math.floor(this.totalPoints));
 55 	}
 56 };
