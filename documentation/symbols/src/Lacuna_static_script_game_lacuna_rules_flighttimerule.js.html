1 /**
  2  * Monitors the flight time of ships and takes action when the time is exceeded
  3  * @class
  4  */
  5 function FlightTimeRule(config){
  6 	RuleBase.call(this, config);
  7 	
  8 	if(!this.maxFlightTime) this.maxFlightTime = 30 * this.engine.timeFactor;
  9 }
 10 
 11 FlightTimeRule.inheritsFrom(RuleBase);
 12 
 13 /**
 14  * Applies the rule to the spaceObjects
 15  * @param {spaceObjects} Array of spaceObjects
 16  * @param {timeLapse} Time that has passed since the last call.
 17  */
 18 FlightTimeRule.prototype.apply = function (spaceObjects, timeLapse) {
 19     var ships = this.engine.level.getShips();
 20     if(ships){
 21 	    for(var i = 0, numberOfShips = ships.length; i < numberOfShips; i++){
 22 	    	var ship = ships[i];
 23 	    	if(ship.status == Ship.statusEnum.flying){
 24 		    	if(!ship.flightTime){
 25 		    		ship.flightTime = timeLapse;
 26 		    	}else{
 27 		    		ship.flightTime += timeLapse;
 28 		    	}
 29 		    	this.engine.level.score.addPoints(10 * timeLapse / this.engine.timeFactor);
 30 		    	if(ship.flightTime > this.maxFlightTime){
 31 		    		ship.setStatus(Ship.statusEnum.crashing);
 32 		    	}
 33 	    	}
 34 	    }
 35     }
 36 };
