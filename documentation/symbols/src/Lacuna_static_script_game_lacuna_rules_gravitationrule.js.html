 14 GravitationRule.prototype.apply = function (spaceObjects, timeLapse) {
 15     if (!spaceObjects) return;
 16 
 17     for (var i = 0, numberOfBodies = spaceObjects.length; i < numberOfBodies; i++) {
 18         var targetBody = spaceObjects[i];
 19         if (targetBody.influencesGravitationalField) {
 20             for (var j = i + 1; j < numberOfBodies; j++) {
 21                 var sourceBody = spaceObjects[j];
 22                 var force = this.getGravitationalForce(targetBody, sourceBody);
 23                 var angleOfForce = targetBody.getAngle(sourceBody);
 24                 var totalWeight = targetBody.mass + sourceBody.mass;
 25 
 26                 if (targetBody.influencedByGravity) {
 27                     var forceMassRatio = force * targetBody.mass / totalWeight;
 28                     var delta = angleOfForce.clone().multiplyScalar(forceMassRatio);
 29                     var newDirection = targetBody.direction.clone().addSelf(delta);
 30 
 31                     targetBody.setDirection(newDirection);
 32                 }
 33 
 34                 // Force also works in the opposite direction ...
 35                 if (sourceBody.influencedByGravity) {
 36                     forceMassRatio = force * sourceBody.mass / totalWeight;
 37                     delta = angleOfForce.clone().negate().multiplyScalar(forceMassRatio);
 38                     newDirection = sourceBody.direction.clone().addSelf(delta);
 39 
 40                     sourceBody.setDirection(newDirection);
 41                 }
 42             }
 43         }
 44     }
 45 };

 47 GravitationRule.prototype.getGravitationalForce = function (object1, object2) {
 48     return this.gravitationalConstant * object1.mass * object2.mass * this.getDistanceGravitationFactor(object1.getDistance(object2));
 49 };
 50 
 51 /* Default but less steep than nature! */
 52 GravitationRule.prototype.getDistanceGravitationFactor = function (distance) {
 53     return 1 / Math.pow(distance, 1.5);
 54 };
