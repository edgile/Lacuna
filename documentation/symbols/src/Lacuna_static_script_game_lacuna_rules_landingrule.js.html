 14 /**
 15  * Applies the rule to the spaceObjects
 16  * @param {spaceObjects} Array of spaceObjects
 17  * @param {timeLapse} Time that has passed since the last call.
 18  * @returns Array of spaceObjects
 19  */
 20 LandingRule.prototype.apply = function (spaceObjects, timeLapse) {
 21     if (!spaceObjects) return;
 22 
 23     var landingZone = this.getLandingZone(spaceObjects);
 24     if (!landingZone) return;
 25 
 26     var newSpaceObjects = [];
 27     var allShips = this.getAllShips(spaceObjects);
 28     if (allShips && allShips.length > 0) {
 29         for (var i = 0, numberOfShips = allShips.length; i < numberOfShips; i++) {
 30             var ship = allShips[i];
 31             var shipStatus = ship.getStatus();
 32 
 33             if (shipStatus == Ship.statusEnum.landing) {
 34                 this.steerShip(ship, landingZone.getPosition());
 35             } else if (shipStatus == Ship.statusEnum.accelerating || shipStatus == Ship.statusEnum.flying) {
 36             	var crossingPoint = this.crossingPoint(ship, landingZone);
 37                 if (crossingPoint) {
 38                 	newSpaceObjects = newSpaceObjects.concat(this.startLandingShip(ship, landingZone, crossingPoint));
 39                 }
 40             }
 41         }
 42     }
 43     return newSpaceObjects;
 44 };

 46 LandingRule.prototype.startLandingShip = function(ship, landingZone, crossingPoint){
 47     ship.setStatus(Ship.statusEnum.landing);
 48     this.level.score.shipLanded();
 49     
 50     var distanceFromOptimum = landingZone.getPosition().distanceTo(crossingPoint);
 51     var points = this.pointsForLanding * (landingZone.halfWidth - distanceFromOptimum) / landingZone.halfWidth; 
 52 	this.level.score.addPoints(points);
 53 	
 54 	return [new TextSpaceObject({text: Math.floor(points), position: ship.getPosition().clone(), direction: ship.getDirection().clone().setLength(500)})];
 55 };

 57 LandingRule.prototype.steerShip = function (ship, targetPosition) {
 58     if (ship.getPosition().distanceTo(targetPosition) < 2) {
 59         ship.setPosition(targetPosition.clone());
 60         ship.setStatus(Ship.statusEnum.landed);
 61     } else {
 62         var directionToHome = new THREE.Vector2();
 63         directionToHome.copy(targetPosition);
 64         var brakePower = Math.max(ship.getVelocity() / 8, 100);
 65         directionToHome.subSelf(ship.getPosition()).setLength(brakePower);
 66 
 67         var newDirection = ship.getDirection().clone().addSelf(directionToHome).multiplyScalar(.95);
 68         ship.setDirection(newDirection);
 69     }
 70 };

 72 /**
 73  * Gets the crossing point of a ship over the landing zone.
 74  * @param {ship} ship to test for a crossing
 75  * @returns null if no crossing happened else coordinates of the crossing point
 76  */
 77 LandingRule.prototype.crossingPoint = function (ship, landingZone) {
 78 	var shipPosition = ship.position.clone();
 79 	var negativeDiretion = ship.getDirection().clone().negate().setLength(5);
 80     var previousPosition = shipPosition.clone().addSelf(negativeDiretion);
 81     var flightPath = new LineSegment(ship.position.clone(), previousPosition);
 82     return landingZone.getLineSegment().intersectSelf(flightPath);
 83 };

 85 LandingRule.prototype.getLandingZone = function (spaceObjects) {
 86     if (!spaceObjects) return;
 87 
 88     for (var i = 0, numberOfObjects = spaceObjects.length; i < numberOfObjects; i++) {
 89         if (spaceObjects[i] instanceof LandingZone) return spaceObjects[i];
 90     }
 91 };

 93 LandingRule.prototype.getAllShips = function (spaceObjects) {

 95 
 96     var result = [];

 98         var object = spaceObjects[i];

102     }
