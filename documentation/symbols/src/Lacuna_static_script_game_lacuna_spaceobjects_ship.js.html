1 /**
  2 *   Represents a Ship.
  3 *   @class
  4 *   @extends SpaceObject
  5 */
  6 var Ship = function(config){
  7 	SpaceObject.call(this, config);
  8 
  9 	// Game logic fields
 10 	this.type = 'ship';
 11 	this.position = this.position || new THREE.Vector2(0,0);
 12     this.maxTailLength = 250;
 13     this.influencedByGravity = true;
 14     this.influencesGravitationalField = false;
 15     this.setStatus(Ship.statusEnum.accelerating);
 16     this.setMass(3000);
 17     this.setDensity(1);
 18     // Visible elements fields 
 19     this.modelIndex = 1;
 20     this.tail = {type: 'polyline', position: {x: 0, y: 0}, width: 2, color: 'yellow', points: new PolyLine()};
 21     this.ship = {type: 'circle', color: 'yellow', center: {x: 0, y: 0}, radius: this.getRadius(), fill: true};
 22     this.flightShapes = [
 23         this.ship,
 24         this.tail
 25     ];
 26     this.crashShapes = [
 27        {type: 'circle', width: 2, color: 'gray', center: {x: 0, y: 0}, radius: 0},
 28        {type: 'circle', width: 2, color: 'gray', center: {x: 0, y: 0}, radius: 0},
 29        {type: 'circle', width: 2, color: 'white', center: {x: 0, y: 0}, radius: 0}
 30 	];
 31     this.shapes = this.flightShapes;
 32 };
 33 
 34 Ship.inheritsFrom(SpaceObject);
 35 
 36 Ship.statusEnum = { finished: 0, accelerating: 1, flying: 2, crashing: 3, landing: 4, landed: 5 };
 37 Ship.statusTimespan = { accelerating: 0, crashing: 5 };
 38 
 39 Ship.prototype.update = function (timeLapse) {
 40     // Update the status of the ship
 41     var status = this.getStatus();
 42     if (status == Ship.statusEnum.accelerating && this.elapsedTime >= Ship.statusTimespan.accelerating) {
 43         this.setStatus(Ship.statusEnum.flying);
 44     }
 45     else if (status == Ship.statusEnum.crashing && this.elapsedTime >= Ship.statusTimespan.crashing) {
 46         this.setStatus(Ship.statusEnum.finished);
 47     }
 48     else if (status == Ship.statusEnum.crashing) {
 49         this.getDirection().setLength(((Ship.statusTimespan.crashing - this.elapsedTime) / Ship.statusTimespan.crashing) * this.getDirection().length());
 50     }
 51     // Update the status of the visuals
 52     if (this.getStatus() == Ship.statusEnum.crashing) {
 53         this.crashShapes[0].radius = Math.max(0, Ship.statusTimespan.crashing - this.elapsedTime) * 2;
 54         this.crashShapes[1].radius = Math.max(0, Ship.statusTimespan.crashing - this.elapsedTime);
 55         this.crashShapes[2].radius = Math.max(0, Ship.statusTimespan.crashing - this.elapsedTime) * 2 / 3;
 56         this.shapes = this.crashShapes;
 57     }
 58     else if (this.getStatus() == Ship.statusEnum.landing) {
 59         this.ship.color = "red";
 60     }
 61     else if (this.getStatus() == Ship.statusEnum.landed) {
 62         this.shapes = [this.ship];
 63         this.ship.color = "green";
 64     }
 65     else if (this.getStatus() == Ship.statusEnum.finished) {
 66         // Do nothing ...
 67     }
 68     else {
 69         this.shapes = this.flightShapes;
 70     }
 71     this.elapsedTime += timeLapse;
 72 };
 73 
 74 Ship.prototype.setPosition = function (position) {
 75     // add previous point to the tail.
 76     if (this.position) {
 77   	  if (this.tail.points.points.length == 0 || (this.tail.points.points.length > 0 && this.position.distanceTo(this.tail.points.points[this.tail.points.points.length - 1]) > 1)) {
 78 	      this.tail.points.addPoint(this.position.clone());
 79 	  }
 80     }
 81     this.baseClass.setPosition.call(this, position);
 82 
 83     // limit the length of the tail.
 84     if (this.maxTailLength > 0) {
 85         while (this.tail.points.getLength() > this.maxTailLength) {
 86             this.tail.points.removePoint(0);
 87         }
 88     }
 89 };
 90 
 91 Ship.prototype.getPreviousPosition = function () {
 92     if (this.tail && this.tail.points.points.length > 1) return this.tail.points.points[this.tail.points.points.length - 1];
 93 };
 94 
 95 Ship.prototype.setStatus = function (status) {
 96     this.baseClass.setStatus.call(this, status);
 97 
 98     this.influencedByGravity = status == Ship.statusEnum.flying || status == Ship.statusEnum.accelerating;
 99     this.canCollide = this.influencedByGravity;
100     this.static = this.status == Ship.statusEnum.landed;
101     this.elapsedTime = 0;
102 };
103 
104 Ship.prototype.getRadius = function () {
105     return 4;
106 };
