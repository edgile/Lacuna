1 /**
  2 Main user interface flow for the game.
  3 @class
  4  */
  5 function flow(config){
  6 	helpers.apply(config, this);
  7 	this.rightButtonDown = false;
  8 	this.engine.keyboardController.upListeners.push(this.keyboardHandler.bind(this));
  9 	this.scoreBar = new ScoreBar({
 10 		engine: this.engine,
 11 		hidden: false
 12 	});
 13 	this.menu = new menu({
 14 		engine: this.engine,
 15 		mainMenu: mainMenu,
 16 		flow: this
 17 	});
 18 	this.start(null, new randomAi({
 19 		engine: this.engine
 20 	}));
 21 	this.menu.show();
 22 }
 23 
 24 flow.prototype.update = function(timeLapse){
 25 	if(this.rightButtonDown != this.engine.buttonDown && this.buttonDown){
 26 		this.menu.toggle();
 27 	}
 28 	this.rightButtonDown = this.engine.buttonDown;
 29 	if(this.ai){
 30 		this.ai.update(timeLapse);
 31 	}
 32 	this.engine.level.update(timeLapse);
 33 	this.menu.update();
 34 };
 35 
 36 flow.prototype.getEntities = function(){
 37 	var gameUI = [];
 38 	if(!this.menu.hidden){
 39 		gameUI.push(this.menu);
 40 	}
 41 	if(!this.scoreBar.hidden){
 42 		gameUI.push(this.scoreBar);
 43 	}
 44 	return this.engine.level.spaceObjects.concat(gameUI);
 45 };
 46 
 47 flow.prototype.keyboardHandler = function(keyCode){
 48 	if(keyCode === 77){ // M
 49 		this.menu.toggle();
 50 	}
 51 };
 52 
 53 flow.prototype.defaultLevel = 'onestarlevel';
 54 
 55 flow.prototype.start = function(levelName, ai){
 56 	this.ai = ai;
 57 	this.levelName = levelName || this.levelName || this.defaultLevel;
 58 	this.engine.level = Levels.load({
 59 		gameId: Constants.gameId,
 60 		levelId: this.levelName,
 61 		engine: this.engine,
 62 		aiMode: this.ai != null
 63 	});
 64 	this.scoreBar.hidden = this.ai != null;
 65 	this.scoreBar.reset();
 66 	this.menu.hide();
 67 };
 68 
 69 // Register with the engine
 70 Engine.games.lacuna = flow;
