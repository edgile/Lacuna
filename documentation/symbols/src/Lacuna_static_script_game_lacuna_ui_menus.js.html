 36 var settingsMenu = {
 37 	'AUDIO': {
 38 		onMousePlaneUp: function (entity, evt) {
 39     		if(Engine.getItem("effectsVolume",'0') === '0'){
 40         		Engine.effectsVolume = 10;
 41         	}
 42         	else{
 43         		Engine.effectsVolume = 0;
 44         	}
 45     		// Persist the setting
 46     		Engine.setItem('effectsVolume', Engine.effectsVolume);
 47     		window.audio.setVolume(Engine.effectsVolume);
 48     		// Update the menu (so it now correctly says 2d/3d)
 49     		entity.engine.flow.menu.setItems(settingsMenu);
 50         },
 51         getText: function(menu){
 52         	var result;
 53         	if(Engine.getItem("effectsVolume",'0') === '0'){
 54         		result = 'AUDIO: OFF';
 55         	}
 56         	else{
 57         		result = 'AUDIO: ON';
 58         	}
 59         	return result; 
 60         }
 61     },
 62     'VIDEO': {
 63     	onMousePlaneUp: function (entity, evt) {
 64     		if(Engine.getItem("renderer",'canvas') === 'canvas'){
 65     			var webglAvailable = ( function () { 
 66     				try { 
 67     					return (!! window.WebGLRenderingContext && !! document.createElement( 'canvas' ).getContext( 'experimental-webgl' )); 
 68 					} catch( e ) { 
 69 						return false; 
 70 					} 
 71 				} )();
 72     			if(webglAvailable){
 73     				Engine.renderer = 'webgl';
 74     			}
 75     			else{
 76     				alert('Sorry, 3D is not support by your bowser.');
 77     			}
 78         	}
 79         	else{
 80         		Engine.renderer = 'canvas';
 81         	}
 82     		// Persist the setting
 83     		Engine.setItem('renderer', Engine.renderer);
 84     		// Update the menu (so it now correctly says 2d/3d)
 85     		entity.engine.flow.menu.setItems(settingsMenu);
 86         },
 87         getText: function(menu){
 88         	var result;
 89         	if(Engine.getItem("renderer",'canvas') === 'canvas'){
 90         		result = 'VIDEO: 2D';
 91         	}
 92         	else{
 93         		result = 'VIDEO: 3D';
 94         	}
 95         	return result; 
 96         }
 97     },
 98     '': {},
 99     'MAIN MENU': {
100         onMousePlaneUp: function (entity, evt) {
101             entity.engine.flow.menu.setItems(mainMenu);
102         }
103     }
104 };
