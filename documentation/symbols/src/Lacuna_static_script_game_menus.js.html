 19 var settingsMenu = {
 20 	'AUDIO': {
 21 		onMousePlaneUp: function (entity, evt) {
 22     		if(engine.getItem("effectsVolume",'0') === '0'){
 23         		engine.effectsVolume = 10;
 24         	}
 25         	else{
 26         		engine.effectsVolume = 0;
 27         	}
 28     		// Persist the setting
 29     		engine.setItem('effectsVolume', engine.effectsVolume);
 30     		window.audio.setVolume(engine.effectsVolume);
 31     		// Update the menu (so it now correctly says 2d/3d)
 32     		entity.engine.rules.menu.setItems(settingsMenu);
 33         },
 34         getText: function(menu){
 35         	var result;
 36         	if(engine.getItem("effectsVolume",'0') === '0'){
 37         		result = 'AUDIO: OFF';
 38         	}
 39         	else{
 40         		result = 'AUDIO: ON';
 41         	}
 42         	return result; 
 43         }
 44     },
 45     'VIDEO': {
 46     	onMousePlaneUp: function (entity, evt) {
 47     		if(engine.getItem("renderer",'classic') === 'classic'){
 48     			var webglAvailable = ( function () { 
 49     				try { 
 50     					return (!! window.WebGLRenderingContext && !! document.createElement( 'canvas' ).getContext( 'experimental-webgl' )); 
 51 					} catch( e ) { 
 52 						return false; 
 53 					} 
 54 				} )();
 55     			if(webglAvailable){
 56     				engine.renderer = 'webgl';
 57     			}
 58     			else{
 59     				alert('Sorry, 3D is not support by your bowser.');
 60     			}
 61         	}
 62         	else{
 63         		engine.renderer = 'classic';
 64         	}
 65     		// Persist the setting
 66     		engine.setItem('renderer', engine.renderer);
 67     		// Update the menu (so it now correctly says 2d/3d)
 68     		entity.engine.rules.menu.setItems(settingsMenu);
 69         },
 70         getText: function(menu){
 71         	var result;
 72         	if(engine.getItem("renderer",'classic') === 'classic'){
 73         		result = 'VIDEO: 2D';
 74         	}
 75         	else{
 76         		result = 'VIDEO: 3D';
 77         	}
 78         	return result; 
 79         }
 80     },
 81     '': {},
 82     'MAIN MENU': {
 83         onMousePlaneUp: function (entity, evt) {
 84             entity.engine.rules.menu.setItems(mainMenu);
 85         }
 86     }
 87 };
