 11 LandingRule.prototype.apply = function (spaceObjects, timeLapse) {
 12     if (!spaceObjects) return;
 13 
 14     var landingPlatform = this.getLandingZone(spaceObjects);
 15     if (!landingPlatform) return;
 16 
 17     var landingShips = this.getLandingShips(spaceObjects);
 18     if (landingShips && landingShips.length > 0) {
 19         for (var i = 0; i < landingShips.length; i++) {
 20             this.steerShips(landingShips, landingPlatform.getPosition());
 21         }
 22     }
 23 
 24     var ships = this.getFlyingShips(spaceObjects);
 25     if (!ships || ships.length == 0) return;
 26 
 27     for (var i = 0; i < ships.length; i++) {
 28         var ship = ships[i];
 29         var shipPreviousPosition = ship.getPreviousPosition();
 30         if (shipPreviousPosition) {
 31             var flightPath = new LineSegment(ship.position.clone(), shipPreviousPosition.clone());
 32             var intersectionPoint = LineSegment.intersect(landingPlatform.getLineSegment(), flightPath);
 33             if (intersectionPoint) {
 34                 ship.setStatus(Ship.statusEnum.landing);
 35             }
 36         }
 37     }
 38 }

 40 LandingRule.prototype.steerShips = function (ships, targetPosition) {
 41     for (var i = 0, numberOfShips = ships.length; i < numberOfShips; i++) {
 42         var ship = ships[i];
 43         if (ship.getPosition().distanceTo(targetPosition) < 2) {
 44             ship.setPosition(targetPosition.clone());
 45             ship.setStatus(Ship.statusEnum.landed);
 46             continue;
 47         }
 48 
 49         var directionToHome = new THREE.Vector2();
 50         directionToHome.copy(targetPosition);
 51         var brakePower = Math.max(ship.getDirection().length() / 8, 100);
 52         directionToHome.subSelf(ship.getPosition()).setLength(brakePower);
 53 
 54         ship.getDirection().addSelf(directionToHome).multiplyScalar(.95);
 55     }
 56 }

 58 LandingRule.prototype.getLandingZone = function (spaceObjects) {
 59     if (!spaceObjects) return;
 60 
 61     for (var i = 0, numberOfObjects = spaceObjects.length; i < numberOfObjects; i++) {
 62         if (spaceObjects[i] instanceof LandingZone) return spaceObjects[i];
 63     }
 64 }

 66 LandingRule.prototype.getFlyingShips = function (spaceObjects) {
 67     if (!spaceObjects) return;
 68 
 69     var result = [];
 70     for (var i = 0, numberOfObjects = spaceObjects.length; i < numberOfObjects; i++) {
 71         var object = spaceObjects[i];
 72         if (object instanceof Ship) {
 73             if (object.getStatus() == Ship.statusEnum.accelerating ||
 74                                     object.getStatus() == Ship.statusEnum.flying) {
 75                 result.push(object);
 76             }
 77         }
 78     }
 79     return result;
 80 }

 82 LandingRule.prototype.getLandingShips = function (spaceObjects) {
 83     if (!spaceObjects) return;
 84 
 85     var result = [];
 86     for (var i = 0, numberOfObjects = spaceObjects.length; i < numberOfObjects; i++) {
 87         var object = spaceObjects[i];
 88         if (object instanceof Ship) {
 89             if (object.getStatus() == Ship.statusEnum.landing) {
 90                 result.push(object);
 91             }
 92         }
 93     }
 94     return result;
